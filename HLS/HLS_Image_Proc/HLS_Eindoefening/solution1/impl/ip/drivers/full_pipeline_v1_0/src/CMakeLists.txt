cmake_minimum_required(VERSION 3.15)
project(full_pipeline)
find_package(common)
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
collect (PROJECT_LIB_SOURCES xfull_pipeline.c)
collect (PROJECT_LIB_SOURCES xfull_pipeline_sinit.c)
collect (PROJECT_LIB_SOURCES xfull_pipeline_linux.c)
collect (PROJECT_LIB_SOURCES xfull_pipeline_g.c)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collect (PROJECT_LIB_HEADERS xfull_pipeline.h)
collect (PROJECT_LIB_HEADERS xfull_pipeline_hw.h)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
include_directories(${CMAKE_BINARY_DIR}/include)
add_library(full_pipeline STATIC ${_sources})
set_target_properties(full_pipeline PROPERTIES LINKER_LANGUAGE C)
